111    def _parse(self, fhs, **kwargs):
112        fhs_list = list(fhs.values())
113        # One element tar.gz -> taxdump.tar.gz
114        if len(fhs_list) == 1 and list(fhs)[0].endswith(".tar.gz"):
115            nodes, ranks, names, self._merged = self._parse_taxdump(
116                fhs_list[0], extended_names=kwargs["extended_names"]
117            )
118        else:
119            # nodes.dmp
120            nodes, ranks = self._parse_nodes(fhs_list[0])
121
122            # [names.dmp]
123            if len(fhs) >= 2:
124                names = self._parse_names(
125                    fhs_list[1], extended_names=kwargs["extended_names"]
126                )
127            else:
128                names = {}
129
130            # [merged.dmp]
131            if len(fhs) == 3:
132                self._merged = self._parse_merged(fhs_list[2])
133        return nodes, ranks, names
134
135    def _parse_merged(self, fh):
136        merged = {}
137        for line in fh:
138            try:
139                old_taxid, _, new_taxid, _ = line.split("\t", 3)
140            except Exception:
141                old_taxid, _, new_taxid, _ = line.decode().split("\t", 3)
142            merged[old_taxid] = new_taxid
143        return merged
144
145    def _parse_names(self, fh, extended_names):
146        names = {}
147        for line in fh:
148            try:
149                node, name, _, name_class = line.split("\t|\t")
150            except Exception:
151                node, name, _, name_class = line.decode().split("\t|\t")
152            if name_class.replace("\t|\n", "") == "scientific name":
153                names[node] = name
154            elif extended_names:
155                if name not in self._extended_name_nodes:
156                    self._extended_name_nodes[name] = []
157                self._extended_name_nodes[name].append(node)
158
159        return names
160
161    def _parse_nodes(self, fh):
162        nodes = {}
163        ranks = {}
164        for line in fh:
165            try:
166                taxid, parent_taxid, rank, _ = line.split("\t|\t", 3)
167            except Exception:
168                taxid, parent_taxid, rank, _ = line.decode().split("\t|\t", 3)
169            ranks[taxid] = rank
170            nodes[taxid] = parent_taxid
171        return nodes, ranks
172
173    def _parse_taxdump(self, fh_taxdump, extended_names):
174        with fh_taxdump.extractfile("nodes.dmp") as fh_nodes:
175            nodes, ranks = self._parse_nodes(fh_nodes)
176        with fh_taxdump.extractfile("names.dmp") as fh_names:
177            names = self._parse_names(fh_names, extended_names=extended_names)
178        with fh_taxdump.extractfile("merged.dmp") as fh_merged:
179            merged = self._parse_merged(fh_merged)
180        return nodes, ranks, names, merged
