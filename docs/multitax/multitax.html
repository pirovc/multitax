191    def _set_root_node(self, root: str, parent: str, name: str, rank: str):
192        """
193        Set root node of the tree.
194        The files are parsed based on the self._default_root_node for each class
195        A user-defined root node can be:
196        1) internal: will filter the tree acodingly and delete the default root_node
197        2) external: will add node and link to the default
198        """
199
200        # Set parent/root with defaults
201        self.root_parent = parent
202        self.root_node = self._default_root_node
203        self._nodes[self.root_node] = self.root_parent
204
205        # Default root node is the top by definition
206        if root != self._default_root_node:
207            if root in self._nodes:
208                # Not default but exists on tree, filter only descendants
209                self.filter(root, desc=True)
210                # Remove entry for _default_root_node
211                self._remove(self._default_root_node)
212            else:
213                # Not on tree, link default node with new root
214                self._nodes[self._default_root_node] = root
215            # Change root to user defined
216            self.root_node = root
217            # Set/Update new root node parent link
218            self._nodes[self.root_node] = self.root_parent
219
220        # User-defined rank/name.
221        # If provided, insert manually,
222        # If None, check if is in the tree (defined in the given tax)
223        #    otherwise insert default "root"
224        if name:
225            self._names[self.root_node] = name
226        elif self.root_node not in self._names:
227            self._names[self.root_node] = "root"
228        # Set static name
229        self.root_name = self._names[self.root_node]
230
231        if rank:
232            self._ranks[self.root_node] = rank
233        elif self.root_node not in self._ranks:
234            self._ranks[self.root_node] = "root"
235        # Set static rank
236        self.root_rank = self._ranks[self.root_node]

313    def check_consistency(self):
314        """
315        Checks consistency of the tree
316
317        Returns: raise an Exception otherwise None
318        """
319        if self.root_node not in self._nodes:
320            raise ValueError("Root node [" + self.root_node + "] not found.")
321        if self.root_parent in self._nodes:
322            raise ValueError(
323                "Root parent ["
324                + self.root_parent
325                + "] found but should not be on tree."
326            )
327        if self.undefined_node in self._nodes:
328            raise ValueError(
329                "Undefined node ["
330                + self.undefined_node
331                + "] found but should not be on tree."
332            )
333
334        # Difference between values and keys should be only root_parent
335        lost_nodes = set(self._nodes.values()).difference(self._nodes)
336        if self.root_parent not in lost_nodes:
337            raise ValueError(
338                "Root parent [" + self.root_parent + "] not properly defined."
339            )
340        # Remove root_parent from lost nodes to report only missing
341        lost_nodes.remove(self.root_parent)
342        if len(lost_nodes) > 0:
343            raise ValueError("Parent nodes missing: " + ",".join(lost_nodes))
344
345        return None

370    def filter(self, nodes: list, desc: bool = False):
371        """
372        Filters taxonomy given a list of nodes.
373        By default keep all the ancestors of the given nodes.
374        If desc=True, keep all descendants instead.
375        Deletes built lineages and translations.
376
377        Example:
378
379            from multitax import GtdbTx
380            tax = GtdbTx()
381
382            tax.lineage('s__Enterovibrio marina')
383            # ['1', 'd__Bacteria', 'p__Proteobacteria', 'c__Gammaproteobacteria', 'o__Enterobacterales', 'f__Vibrionaceae', 'g__Enterovibrio', 's__Enterovibrio marina']
384            # Keep only ancestors of 'g__Enterovibrio'
385            tax.filter('g__Enterovibrio')
386
387            # Reload taxonomy
388            tax = GtdbTx()
389            # Keep only descendants of 'g__Enterovibrio'
390            tax.filter('g__Enterovibrio', desc=True)
391        """
392        if isinstance(nodes, str):
393            nodes = [nodes]
394
395        # Keep track of nodes to be filtered out
396        filtered_nodes = set(self._nodes)
397        # Always keep root
398        filtered_nodes.discard(self.root_node)
399
400        if desc:
401            # Keep descendants of the given nodes
402            for node in nodes:
403                # Check if node exists (skips root)
404                if node in filtered_nodes:
405                    # For each leaf of the selected nodes
406                    for leaf in self.leaves(node):
407                        # Build lineage of each leaf up-to node itself
408                        for n in self.lineage(leaf, root_node=node):
409                            # Discard nodes from set to be kept
410                            filtered_nodes.discard(n)
411                    # Link node to root
412                    self._nodes[node] = self.root_node
413        else:
414            # Keep ancestors of the given nodes (full lineage up-to root)
415            for node in nodes:
416                # ranks=[] in case build_lineages() was used with specific ranks
417                for n in self.lineage(node, ranks=[]):
418                    # Discard nodes from set to be kept
419                    filtered_nodes.discard(n)
420
421        # Delete filtered nodes
422        for node in filtered_nodes:
423            self._remove(node)
424
425        # Delete aux. data structures
426        self._reset_aux_data()
427        self.check_consistency()

452    def lineage(self, node: str, root_node: str = None, ranks: list = None):
453        """
454        Returns a list with the lineage of a given node.
455        If ranks is provided, returns only nodes annotated with such ranks.
456        If root_node is provided, use it instead of default root of tree.
457        """
458        # If lineages were built with build_lineages() with matching params
459        if node in self._lineages and root_node is None and ranks is None:
460            return self._lineages[node]
461        else:
462            if not root_node:
463                root_node = self.root_node
464
465            n = node
466            if ranks:
467                # Fixed length lineage
468                lin = [self.undefined_node] * len(ranks)
469                # Loop until end of the tree (in case chosen root is not on lineage)
470                while n != self.undefined_node:
471                    r = self.rank(n)
472                    if r in ranks:
473                        lin[ranks.index(r)] = n
474                    # If node is root, break (after adding)
475                    if n == root_node:
476                        break
477                    n = self.parent(n)
478            else:
479                # Full lineage
480                lin = []
481                # Loop until end of the tree (in case chosen root is not on lineage)
482                while n != self.undefined_node:
483                    lin.append(n)
484                    # If node is root, break (after adding)
485                    if n == root_node:
486                        break
487                    n = self.parent(n)
488                # Reverse order
489                lin = lin[::-1]
490
491            # last iteration node (n) != root_node: didn't find the root, invalid lineage
492            if n != root_node:
493                return []
494            else:
495                return lin

542    def prune(self, nodes: list):
543        """
544        Prunes branches of the tree under the given nodes.
545        Deletes built lineages and translations.
546        """
547
548        if isinstance(nodes, str):
549            nodes = [nodes]
550
551        del_nodes = set()
552        for node in nodes:
553            if node not in self._nodes:
554                raise ValueError("Node [" + node + "] not found.")
555            for leaf in self.leaves(node):
556                for n in self.lineage(leaf, root_node=node)[1:]:
557                    del_nodes.add(n)
558
559        for n in del_nodes:
560            self._remove(n)

562        self._reset_aux_data()

666        self,
667        output_file: str,
668        cols: list = ["node", "parent", "rank", "name"],
669        sep: str = "\t",
670        sep_multi: str = "|",
671        ranks: list = None,
672        gz: bool = False,
673    ):
674        """
675        Writes loaded taxonomy to a file.
676
677        Parameters:
678        * **cols** *[list]*: Options: "node", "latest", "parent", "rank", "name", "leaves", "children", "lineage", "rank_lineage", "name_lineage"
679        * **sep** *[str]*: Separator of fields
680        * **sep_multi** *[str]*: Separator of multi-valued fields
681        * **ranks** *[list]*: Ranks to report
682        * **gz** *[bool]*: Gzip output
683
684        Returns: None
685        """
686        import gzip
687
688        if gz:
689            output_file = (
690                output_file if output_file.endswith(".gz") else output_file + ".gz"
691            )
692            check_no_file(output_file)
693            outf = gzip.open(output_file, "wt")
694        else:
695            check_no_file(output_file)
696            outf = open(output_file, "w")
697
698        write_field = {
699            "node": lambda node: node,
700            "latest": self.latest,

453    def lineage(self, node: str, root_node: str = None, ranks: list = None):
454        """
455        Returns a list with the lineage of a given node.
456        If ranks is provided, returns only nodes annotated with such ranks.
457        If root_node is provided, use it instead of default root of tree.
458        """
459        # If lineages were built with build_lineages() with matching params

462        else:

464                root_node = self.root_node
465
466            n = node
467            if ranks:
468                # Fixed length lineage
469                lin = [self.undefined_node] * len(ranks)
470                # Loop until end of the tree (in case chosen root is not on lineage)
471                while n != self.undefined_node:
472                    r = self.rank(n)
473                    if r in ranks:
474                        lin[ranks.index(r)] = n

479            else:
480                # Full lineage

486                    if n == root_node:

488                    n = self.parent(n)
489                # Reverse order
490                lin = lin[::-1]
491
